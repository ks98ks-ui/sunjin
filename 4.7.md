mkdir tutoriall 폴더만들기
cd tutoriall 폴더이동
git init 
echo "원숭이 선생님의 Git 명령" > myfile.txt 텍스트만들기
git add . 
git status 정보확인
git commit -m "first commit" 첫번째 커밋만들기
git branch issue1 이슈1 만들기 git branch 확인 git checkout issue1 이슈1로 이동
echo "add 변경을 인덱스에 등록" >>myfile.txt 내용추가 cat 내용확인
git add myfile.txt 텍스트에 추가 git commit -m "add설명추가" 커밋추가
*branch 이동시 commit 하고 이동할것*
git checkout master 마스터로 이동후 git merge issue1 이슈1 병합후 git branch -d issue1 이슈1삭제
git branch issue3 git checkout -b issue2 이슈3생성, 이슈2 생성후 이슈2로 이동
echo "commit 인덱스의 상태를 기록">>myfile.txt 내용추가 git add . 내용추가 git commit -m "commit 설명추가"
커밋후 git checkout issue3 이슈3로 이동 
echo "pull 원격 리포지토리의 내용을 가져옵니다" >>myfile.txt 내용추가 git add . 내용추가 git commit -m "pull 설명추가"
커밋후 git checkout master 마스터로 이동
git merge issue2 이슈2를 마스터로 합침 그후 git merge issue3 합치면 오류발생
파일을 열어 직접 수정 한 후 git add . 후 git commit -m 커밋추가후 로그 확인 :탈출 q
rebase로 정리하는법 git reset --hard HEAD~ 한단계전으로 리셋 git checkout issue3
git rebase master 후 직접수정후 저장, esc :q 엔터로 탈출
